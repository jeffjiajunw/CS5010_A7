# this sample file runs the new commands offered by the IME program on the given sample image.
# (new as of assignment 5)

# we can load images with the "load <path> <name>" command.
load res/bambi.png bambi
save res/bambi-copy.png bambi

# we can generate a histogram of our images using the "histogram <src-image-name> <dest-image-name>" command.
histogram bambi bambi-histogram

# we can compress images with the "compress <percentage> <src-image-name> <dest-image-name>" command.
compress 20 bambi bambi-compressed-20
compress 95 bambi bambi-compressed-95
histogram bambi-compressed-95 bambi-compressed-95-histogram

# we can color-correct an image with the "color-correct <image-name> <dest-image-name>"
color-correct bambi bambi-color-corrected
color-correct bambi bambi-color-corrected-split split 50
histogram bambi-color-corrected bambi-color-corrected-histogram

# we can do levels-adjustment using the "levels <image-name> <dest-image-name> [b] [m] [w]" command.
levels bambi bambi-levels 20 100 255
levels bambi bambi-levels-2 30 80 240
levels bambi bambi-levels-split 20 100 255 split 50
levels bambi bambi-levels-split 20 100 255 split 50
histogram bambi-levels bambi-levels-histogram

# we can save all the images we created so far using the "save <path> <name>" command.

save res/bambi-histogram.png bambi-histogram

save res/bambi-compressed-20.png bambi-compressed-20
save res/bambi-compressed-95.png bambi-compressed-95
save res/bambi-compressed-95-histogram.png bambi-compressed-95-histogram

save res/bambi-color-corrected.png bambi-color-corrected
save res/bambi-color-corrected-split.png bambi-color-corrected-split
save res/bambi-color-corrected-histogram.png bambi-color-corrected-histogram

save res/bambi-levels.png bambi-levels
save res/bambi-levels-2.png bambi-levels-2
save res/bambi-levels-split.png bambi-levels-split
save res/bambi-levels-histogram.png bambi-levels-histogram
